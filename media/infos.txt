Table trips_gtfs_rt {
  trip_id varchar pk
  trip_headline int [ref: - trips_gtfs.trip_headline]
  departure_date date
  departure_time time
}


Table stop_time_update_gtfs_rt {
  trip_id varchar pk [ref: > trips_gtfs_rt.trip_id]
  stop_id VARCHAR [ref: - stops_gtfs.stop_id]
  arrival_time TIMESTAMP
  departure_time TIMESTAMP
  delay_arrival INT
  delay_departure INT
}


Table trips_gtfs {
  trip_headline int pk
  route_id varchar [ref: - routes_gtfs.route_id]
  trip_id varchar [ref: < stop_times_gtfs.trip_id]
  direction_id int
}


Table routes_gtfs {
  route_id varchar pk
  route_short_name varchar
  route_long_name varchar
  route_type int
  location_type int
}


Table stop_times_gtfs {
  trip_id varchar pk
  arrival_time time
  departure_time time
  stop_id varchar [ref: > stops_gtfs.stop_id]
  stop_sequence int
  pickup_type int
  drop_off_type int
}


Table stops_gtfs {
  stop_id varchar pk
  stop_name varchar
  stop_lat float
  stop_lon float
  location_type int
}
